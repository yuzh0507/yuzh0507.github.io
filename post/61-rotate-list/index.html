61. Rotate List

2022-06-12 2 min read # C++ # LeetCode

Problem
Given the head of a linked list, rotate the list to the right by k places.

Example

Input: head = [1,2,3,4,5], k = 2
Output: [4,5,1,2,3]

Constraints

  • The number of nodes in the list is in the range [0, 500].
  • -100 <= Node.val <= 100
  • 0 <= k <= 2 * 109

Link: https://leetcode.com/problems/rotate-list/

Description

  1. Build a new ListNode, value = last.value, next = head
  2. Refresh the head and tail
  • Time Complexity: O(n)
  • Space Complexity: O(n)

Code

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
private:
    int ListLength(ListNode * head)
    {
        int length = 0;
        ListNode * tmp = head;
        for(; tmp != nullptr; tmp = tmp->next)
            length++;

        return length;
    }
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if(head == nullptr)
            return nullptr;
        
        if(head->next == nullptr)
            return head;

        // only k % length is necessary 
        k = k % ListLength(head);
        while(k>0)
        {
            ListNode * newHead = new ListNode;
            ListNode * tail = head;
            ListNode * beforeTail;
            while(tail->next != nullptr)
            {
                if(tail->next->next == nullptr)
                    beforeTail = tail;
                tail = tail->next;
            }
            newHead->val = tail->val;
            newHead->next = head;
            head = newHead;
            beforeTail->next = nullptr;

            k--;
        }

        return head;
    }
};
