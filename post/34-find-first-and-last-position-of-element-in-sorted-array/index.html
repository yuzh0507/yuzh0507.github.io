34. Find First and Last Position of Element in Sorted Array

2022-05-21 2 min read # C++ # LeetCode

Problem

Given an array of integers nums sorted in non-decreasing order, find the starting and ending position of a given target value.
If target is not found in the array, return [-1, -1].
You must write an algorithm with O(log n) runtime complexity.

Example

Input: nums = [5,7,7,8,8,10], target = 8
Output: [3,4]

Input: nums = [5,7,7,8,8,10], target = 6
Output: [-1,-1]

Input: nums = [], target = 0
Output: [-1,-1]

Constraints

0 <= nums.length <= 105
-109 <= nums[i] <= 109
nums is a non-decreasing array.
-109 <= target <= 109

Link: https://leetcode.com/problems/find-first-and-last-position-of-element-in-sorted-array/

Description

  1. binary search the target.
  2. If searched, find the first and last element position.

Code

class Solution {
public:
    vector<int> searchRange1(vector<int>& nums, int target)
    {
        if (nums.empty())
            return {-1, -1};
        if (nums.size() == 1)
            return {nums[0]==target ? 0:-1, nums[0]==target ? 0:-1};

        vector<int> pos;
        bool found = false;

        int left = 0, right = nums.size()-1, mid = (left+right)/2;
        for(; left <= right; mid = (left+right)/2)
        {
            if(target < nums[mid])
            {
                right = mid-1;
            }
            else if (target > nums[mid])
            {
                left = mid+1;
            }
            else
            {
                found = true;
                break;
            }
        }

        if(!found)
        {
            pos.push_back(-1);
            pos.push_back(-1);
        }
        else
        {
            int first = mid, last = mid;
            if(first > 0)
            while(first > 0 && nums[first-1] == target)
            {
                first--;
            }
            while(last < nums.size()-1 && nums[last+1] == target)
            {
                last++;
            }
            
            pos.push_back(first);
            pos.push_back(last);
        }

        return pos;
    }
};
