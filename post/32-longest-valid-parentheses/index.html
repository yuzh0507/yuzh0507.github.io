32. Longest Valid Parentheses

2022-05-19 2 min read # C++ # LeetCode

Problem

Given a string containing just the characters '(' and ')', find the length of the longest valid (well-formed) parentheses substring.

Example

Input: s = "(()"
Output: 2
Explanation: The longest valid parentheses substring is "()".

Input: s = ")()())"
Output: 4
Explanation: The longest valid parentheses substring is "()()".

Input: s = ""
Output: 0

Constraints

0 <= s.length <= 3 * 104
s[i] is '(', or ')'.

Link: https://leetcode.com/problems/longest-valid-parentheses/submissions/

Description

  1. Use a C++ stack to store the index of s.
  2. if (s[i] == '('), st.push(i)
  3. if (s[i] == ')'),
    3.1. if(st.empty() || st.top() == ')'), st.push(i), num = 0;
    3.2. else, st.pop(), num = i - (st.empty() ? -1 : st.top()); longest = max(num, longest)

Code

class Solution {
private:
    int longestValidParentheses(string s)
    {
        if(s.size() < 2)
            return 0;
        
        int num = 0;
        int longest = 0;
        stack<int> st;
        for(int i = 0; i < s.size(); i++)
        {
            if(s[i] == '(')
            {
                st.push(i);
            }

            else
            {
                if(st.empty() || s[st.top()] == ')')
                {
                    num = 0;
                    st.push(i);
                }
                else
                {
                    st.pop();
                    num = i - (st.empty() ? -1 : st.top());
                    longest = max(longest, num);
                }
            }
        }
            
        return longest;
    }
};
