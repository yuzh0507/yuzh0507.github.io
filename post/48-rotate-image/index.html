48. Rotate Image

2022-06-03 2 min read # C++ # LeetCode

Problem
You are given an n x n 2D matrix representing an image, rotate the image by 90 degrees (clockwise).

You have to rotate the image in-place, which means you have to modify the input 2D matrix directly. DO NOT allocate another 2D matrix and do the rotation.

Example

Input: matrix = [[1,2,3],[4,5,6],[7,8,9]]
Output: [[7,4,1],[8,5,2],[9,6,3]]

Input: matrix = [[5,1,9,11],[2,4,8,10],[13,3,6,7],[15,14,12,16]]
Output: [[15,13,2,5],[14,3,4,1],[12,6,8,9],[16,7,10,11]]

Constraints

  • n == matrix.length == matrix[i].length
  • 1 <= n <= 20
  • -1000 <= matrix[i][j] <= 1000

class Solution {
private:
    void rotateElem(vector<vector<int>>& matrix, const int & n, int row, int col)
    {
        if(col == n-1-row)
        {
            row++;
            col = row;
        }

        int tmp1 = matrix[col][n-1-row];
        matrix[col][n-1-row] = matrix[row][col];

        int tmp2 = matrix[n-1-row][n-1-col];
        matrix[n-1-row][n-1-col] = tmp1;

        matrix[row][col] = matrix[n-1-col][row];

        matrix[n-1-col][row] = tmp2;

        if (row == col && row == (n-1)/2)
            return;
        else
            rotateElem(matrix, n, row, col+1);
    }
public:
    void rotate(vector<vector<int>>& matrix) {
        if (matrix.size() < 2)
            return;
        
        rotateElem(matrix, matrix.size(), 0, 0);
    }
};
