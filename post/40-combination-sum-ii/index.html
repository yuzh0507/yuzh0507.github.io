40. Combination Sum II

2022-05-27 3 min read # C++ # LeetCode

Problem

Given a collection of candidate numbers (candidates) and a target number (target), find all unique combinations in candidates where the candidate numbers sum to target.
Each number in candidates may only be used once in the combination.
Note: The solution set must not contain duplicate combinations.

Example

Input: candidates = [10,1,2,7,6,1,5], target = 8
Output: 
[
[1,1,6],
[1,2,5],
[1,7],
[2,6]
]

Input: candidates = [2,5,2,1,2], target = 5
Output: 
[
[1,2,2],
[5]
]

Constraints

1 <= candidates.length <= 100
1 <= candidates[i] <= 50
1 <= target <= 30

Link: https://leetcode.com/problems/combination-sum-ii/

Description

  1. Sort the candidates
  2. Recursively call thesum(candidates, target - candidates[i], i+1, elem, ans)

Code

class Solution {
private:
    void sum (vector<int>& candidates, int target, int i, vector<int>& elem, vector<vector<int>>& ans)
    {
        // If target = 0, the current elem is valid, push back into ans
        if(target == 0)
        {
            ans.push_back(elem);
            return;
        }

        // 
        while(i < candidates.size() && target - candidates[i] >= 0)
        {
            // Recursively candidates[i] into elem, until target - candidates[i] < 0
            elem.push_back(candidates[i]);        
            sum(candidates, target - candidates[i], i+1 , elem, ans);

            // Once return from the recursion, pop_back the last number of elem
            elem.pop_back();
            
            // If candidates[i+1] == candidates[i], keep i++
            while (i+1 < candidates.size() && candidates[i+1] == candidates[i])
                i++;
            i++;
        }
    }
    
public:
    vector<vector<int>> combinationSum2(vector<int>& candidates, int target) {
        sort(candidates.begin(), candidates.end());

        vector<int> elem;
        vector<vector<int>> ans;
        sum(candidates, target, 0, elem, ans);

        return ans;
    }
};
