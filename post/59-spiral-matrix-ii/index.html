59. Spiral Matrix II

2022-06-11 3 min read # C++ # LeetCode

Problem
Given a positive integer n, generate an n x n matrix filled with elements from 1 to n2 in spiral order.

Example

Input: n = 3
Output: [[1,2,3],[8,9,4],[7,6,5]]

Constraints

  • 1 <= n <= 20

Link: https://leetcode.com/problems/spiral-matrix-ii/

Description
Recursively fill numbers: right -> down -> left -> up -> right ...

  • Time Complexity: O(n)
  • Space Complexity: O(n)

Code

class Solution {
private:
    void fillRight(vector<vector<int>> & op, int row, int col, int rightBound, int value)
    {
        if(col > rightBound)
            return;

        for(int i = col; i <= rightBound; i++)
        {
            op[row][i] = value++;
        }
        fillDown(op, row+1, rightBound, op[row].size()-row-1, value);
    }

    void fillDown(vector<vector<int>> & op, int row, int col, int downBound, int value)
    {
        if(row > downBound)
            return;

        for(int i = row; i <= downBound; i++)
        {
            op[i][col] = value++;
        }

        fillLeft(op, downBound, col-1, op[0].size()-col-1, value);

    }

    void fillLeft(vector<vector<int>> & op, int row, int col, int leftBound, int value)
    {
        if(col < leftBound)
            return;

        for(int i = col; i >= leftBound; i--)
        {
            op[row][i] = value++;
        }

        fillUp(op, row-1, leftBound, op.size()-row, value);
    }

    void fillUp(vector<vector<int>> & op, int row, int col, int upBound, int value)
    {
        if(row < upBound)
            return;

        for(int i = row; i >= upBound; i--)
        {
            op[i][col] = value++;
        }

        fillRight(op, upBound, col+1, op[0].size()-col-2, value);
    }
public:
    vector<vector<int>> generateMatrix(int n) {
        vector<vector<int>> op(n,vector<int>(n, 0));
        fillRight(op, 0, 0, n-1, 1);

        return op;
    }
};
