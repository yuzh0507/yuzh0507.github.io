82. Remove Duplicates from Sorted List II

2022-07-02 3 min read # C++ # LeetCode

Problem
Given the head of a sorted linked list, delete all nodes that have duplicate numbers, leaving only distinct numbers from the original list. Return the linked list sorted as well.

Example

Input: head = [1,2,3,3,4,4,5]
Output: [1,2,5]

Input: head = [1,1,1,2,3]
Output: [2,3]

Constraints

  • The number of nodes in the list is in the range [0, 300].
  • -100 <= Node.val <= 100
  • The list is guaranteed to be sorted in ascending order.

Link: https://leetcode.com/problems/remove-duplicates-from-sorted-list-ii/

Description
Binary search

  1. If (head->next->val = head->val), move *head to right
  2. build two pointer "*left" and "*right", use *right to check duplicates, use left to change pointers
  • Time Complexity: O(n)
  • Space Complexity: O(1)

Code

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head == nullptr || head->next == nullptr)
            return head;

        while (head->next->val == head->val)
        {
            while(head->next != nullptr && head->next->val == head->val)
            {
                head = head->next;
            }
            head = head->next;

            if(head == nullptr || head->next == nullptr)
                break;
        }

        ListNode* left = head;
        if(left == nullptr)
            return nullptr;

        ListNode* right = head->next;

        while(right != nullptr && right->next != nullptr)
        {
            if(right->next->val == right->val)
            {
                while(right->next != nullptr && right->next->val == right->val)
                {
                    right = right->next;
                }
                left->next = right->next;
                right = left->next;
            }
            else
            {
                left = right;
                right = left->next;
            }
        }

        return head;
    }
};
