24. Swap Nodes in Pairs

2022-05-15 2 min read # C++ # LeetCode

Problem

Given a linked list, swap every two adjacent nodes and return its head. You must solve the problem without modifying the values in the list's nodes (i.e., only nodes themselves may be changed.)

Example

Input: head = [1,2,3,4]
Output: [2,1,4,3]

Constraints

The number of nodes in the list is in the range [0, 100].
0 <= Node.val <= 100

Link: https://leetcode.com/problems/swap-nodes-in-pairs/

Description

Code

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:    
    ListNode* swapPairs(ListNode* head)
    {
        if(head == nullptr)
            return nullptr;
        if(head->val != 0 && head->next == nullptr)
            return head;

        ListNode * tmp = head;
        // Build a new ListNode, assign value and *next for it
        ListNode * n = new ListNode();
        n->val = head->val;
        n->next = head->next->next;

        // Connect the second ListNode to the *n
        head->next->next = n;

        // Reassign *head
        head = head->next;

        // *tmp point to *n
        tmp = n;
        
        while (tmp->next != nullptr)
        {
            if(tmp->next->next != nullptr)
            {
                // Build a new ListNode, assign value and *next for it
                n = new ListNode();
                n->val = tmp->next->val;
                n->next = tmp->next->next->next;

                // Connect the second ListNode to the *n
                tmp->next->next->next = n;

                // Connect the *tmp to the second ListNode
                tmp->next = tmp->next->next;

                // *tmp point to *n
                tmp = n;
            }
            else
            {
                tmp = tmp->next;
            }
        }
        return head;
    }
};
