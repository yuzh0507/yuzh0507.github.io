5. Longest Palindromic Substring

2022-05-06 2 min read # C++ # LeetCode

Problem

Given a string s, return the longest palindromic substring in s.

Constraints

1 <= s.length <= 1000
s consist of only digits and English letters.

Link: https://leetcode.com/problems/longest-palindromic-substring/

class Solution {
public:
    string longestPalindrome(string s) {
        int n = s.size();
        if(n<2)
            return s;
        
        int begin = 0;
        int maxLen = 1;

        // Build a matrix PalindromicMatrix to record whether s[i]~s[j] is a palindromic substring
        vector<vector<bool>> PM(n,vector<bool>(n));     
        
        // PM Initialization
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                    PM[i][j] == true;
                else
                    PM[i][j] == false;
            }
            
        }
        
        // Enumerate substring length
        for (int L = 1; L < n; L++)
        {
            // Start from the left side
            for(int left = 0; left+L < n; left++)
            {
                if (s[left] != s[left+L])
                    PM[left][left+L] = false;
                    

                else if (L<=2)
                    PM[left][left+L] = true;
                    
                else
                    // if (s[i] == s[j]) && (s[i+1] == s[j-1]), the bigger substring is palindromic  
                    PM[left][left+L] = PM[left+1][left+L-1];
                
                // Record the position of longest palindromic substring 
                if(PM[left][left+L] && maxLen < L+1)
                {
                    begin = left;
                    maxLen = L+1;
                }
            }
        }
        return s.substr(begin, maxLen);
    }
};
